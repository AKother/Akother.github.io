import java.io.*;
import java.util.*;

// 类名称必须采用public class Main方式命名
public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(new BufferedInputStream(System.in)); // 更快一点
        // Scanner scanner = new Scanner(System.in);
        int a = scanner.nextInt();          // 读入数字
        double b = scanner.nextDouble();    // 读入浮点数
        String s = scanner.next();          // 读入字符串

        // 判断是否有下一个输入
        scanner.hasNextInt();
        scanner.hasNextDouble();
        scanner.hasNext();

        // 输出函数
        System.out.print(a);
        System.out.println(b);
        System.out.format("%.2f, %s", 1.0 * a / b, s);


        // example1 : 读入多个整数
        while(scanner.hasNext()){
            int score = scanner.nextInt();
        }

        // example2 : 已知字符串数目，读入多行字符串
        int num = scanner.nextInt();
        for (int i = 0; i < num; i++) {
            String str = scanner.next();
        }

        // example3 : 读取二维数组
        while(scanner.hasNext()){
            int r = scanner.nextInt();
            int c = scanner.nextInt();
            int[][] matrix = new int[r][c];
            scanner.nextLine(); // 跳换行符
            for(int i = 0; i < r; i++){
                for(int j = 0; j < c; j++){
                    matrix[i][j] = scanner.nextInt();
                }
            }
        }
    }
}
